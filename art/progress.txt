    // 3. DATA LOGIC
    // =================================================================

    d3.csv("artDataset.csv").then(raw_data => {
        document.getElementById("loading").style.display = "none";

        const counts = {};
        raw_data.forEach(d => { if (d.movement) counts[d.movement.trim()] = (counts[d.movement.trim()] || 0) + 1; });
        const topMovements = Object.keys(counts).sort((a, b) => counts[b] - counts[a]).slice(0, 5);
        const finalMovementsList = [...topMovements, "Others"];

        const colorScale = d3.scaleOrdinal(BUBBLE_COLORS).domain(finalMovementsList);

        const data = raw_data.map((d, i) => {
            let cleanPrice = d.price ? parseFloat(d.price.replace(/[^\d]/g, '')) : 0;
            let cleanYear = d.yearCreation ? parseInt(d.yearCreation.match(/\d{4}/)) : null;
            let rawMov = d.movement ? d.movement.trim() : "Unknown";
            let finalMov = topMovements.includes(rawMov) ? rawMov : "Others";
            return { id: i, title: d.title, artist: d.artist, movement: finalMov, price: cleanPrice, year: cleanYear };
        }).filter(d => d.year && d.year > 1400 && d.price > 0);

        // UPDATED: Much larger bubbles (20px to 70px) to fill space
        const priceScale = d3.scaleSqrt().domain(d3.extent(data, d => d.price)).range([20, 70]);

        const svg = d3.select("#viz-container").append("svg")
            .attr("width", window.innerWidth)
            .attr("height", window.innerHeight)
            .attr("viewBox", [-window.innerWidth / 2, -window.innerHeight / 2, window.innerWidth, window.innerHeight]);

        const tooltip = d3.select("#tooltip");
        const paletteLayer = svg.append("g").attr("class", "palette-layer");
        const blobLayer = svg.append("g").attr("class", "blob-layer");
        const splatterLayer = svg.append("g").attr("class", "splatter-layer").style("opacity", 0).style("display", "none");
        const dataLayer = svg.append("g").attr("class", "data-layer");
        const interactionLayer = svg.append("g").attr("class", "interaction-layer");

        paletteLayer.append("image")
            .attr("href", "transPalette.png") 
            .attr("width", VISUAL_WIDTH).attr("height", VISUAL_HEIGHT) 
            .attr("x", -VISUAL_WIDTH / 2).attr("y", -VISUAL_HEIGHT / 2); 

        const splatterShape = splatterLayer.append("path")
            .attr("class", "splatter-shape")
            .style("stroke", "none");

        const bubbles = dataLayer.selectAll(".data-bubble").data(data).enter().append("circle")
            .attr("class", "data-bubble").attr("r", d => priceScale(d.price)).attr("fill", d => colorScale(d.movement))
            .attr("opacity", 0).style("display", "none")
            .on("mouseover", (e, d) => tooltip.style("opacity", 1).html(`<strong>${d.title}</strong><br>${d.artist}<br>${d.year}<br>$${d.price.toLocaleString()}`))
            .on("mousemove", (event) => tooltip.style("left", (event.pageX + 15) + "px").style("top", (event.pageY - 15) + "px"))
            .on("mouseout", () => tooltip.style("opacity", 0))
            .on("click", handleArtistTrace);

        const movementGlobs = finalMovementsList.map((m, i) => {
            const pos = PAINT_LOCATIONS[i] || { x: 0, y: 0 };
            return { name: m, x: pos.x, y: pos.y, img: BLOB_IMAGES[i], pathIndex: i % 6 };
        });

        blobLayer.selectAll(".paint-blob-img").data(movementGlobs).enter().append("image")
            .attr("xlink:href", d => d.img).attr("width", BLOB_IMG_SIZE).attr("height", BLOB_IMG_SIZE)
            .attr("x", (d, i) => d.x - (BLOB_IMG_SIZE / 2) + BLOB_NUDGES[i].x) 
            .attr("y", (d, i) => d.y - (BLOB_IMG_SIZE / 2) + BLOB_NUDGES[i].y);

        interactionLayer.selectAll(".movement-glob").data(movementGlobs).enter().append("circle")
            .attr("r", INTERACTION_RADIUS).attr("cx", d => d.x).attr("cy", d => d.y)
            .attr("fill", "transparent").attr("pointer-events", "all")
            .style("cursor", "pointer")
            .on("mouseover", function (event, d) {
                tooltip.style("opacity", 1).html(`<strong>${d.name}</strong><br>${counts[d.name] || 'Mixed'} artworks`);
                d3.select(this).on("mousemove", (e) => tooltip.style("left", (e.pageX + 15) + "px").style("top", (e.pageY + 15) + "px"));
            })
            .on("mouseout", () => tooltip.style("opacity", 0))
            .on("click", (event, d) => enterSplatterMode(d));

        let simulation = d3.forceSimulation(data)
            .force("collide", d3.forceCollide().radius(d => priceScale(d.price) + 1).strength(1))
            .stop();

        // =================================================================
        // ENTER SPLATTER MODE
        // =================================================================
        function enterSplatterMode(selectedGlob) {
            const selectedMovement = selectedGlob.name;
            const selectedPath = SPLATTER_PATHS[selectedGlob.pathIndex];
            const baseColor = colorScale(selectedMovement);

            paletteLayer.transition().duration(500).style("opacity", 0);
            blobLayer.transition().duration(500).style("opacity", 0);
            interactionLayer.style("display", "none"); 

            splatterShape.attr("d", selectedPath).attr("transform", null);
            
            const bbox = splatterShape.node().getBBox();
            const pathCenterX = bbox.x + bbox.width / 2;
            const pathCenterY = bbox.y + bbox.height / 2;
            const scaleFactor = (Math.min(window.innerWidth, window.innerHeight) * 0.85) / Math.max(bbox.width, bbox.height);

            // FLIP TRANSFORM LOGIC
            let extraTransform = "";
            if (selectedGlob.pathIndex === 3) extraTransform = "scale(-1, 1)"; // Flip Mint
            if (selectedGlob.pathIndex === 5) extraTransform = "scale(-1, -1)"; // Flip Blue

            splatterShape
                .attr("transform", `scale(${scaleFactor}) ${extraTransform} translate(${-pathCenterX}, ${-pathCenterY})`)
                .style("fill", baseColor)
                .style("opacity", 0.2);

            splatterLayer.style("display", "block").transition().duration(1000).style("opacity", 1);
            document.getElementById("resetBtn").style.display = "block"; 
            tooltip.style("opacity", 0);
            
            const activeData = data.filter(d => d.movement === selectedMovement);
            bubbles.filter(d => d.movement === selectedMovement).style("display", "block").transition().duration(800).attr("opacity", 0.9);
            bubbles.filter(d => d.movement !== selectedMovement).style("display", "none").attr("opacity", 0);

            // SAMPLER (Canvas check)
            const samplerCanvas = document.createElement("canvas");
            samplerCanvas.width = window.innerWidth; samplerCanvas.height = window.innerHeight;
            const ctx = samplerCanvas.getContext("2d");
            
            ctx.translate(window.innerWidth/2, window.innerHeight/2); 
            ctx.scale(scaleFactor, scaleFactor);
            if (selectedGlob.pathIndex === 3) ctx.scale(-1, 1);
            if (selectedGlob.pathIndex === 5) ctx.scale(-1, -1);
            ctx.translate(-pathCenterX, -pathCenterY);
            
            const p2d = new Path2D(selectedPath);

            activeData.forEach(d => {
                let found = false, attempts = 0;
                // Increase attempts to 1000 to ensure we find the edges
                while(!found && attempts < 1000) {
                    const randX = bbox.x + Math.random() * bbox.width;
                    const randY = bbox.y + Math.random() * bbox.height;
                    
                    // Check validity in SVG Space
                    if(ctx.isPointInPath(p2d, randX, randY)) {
                        // Convert to Screen Coordinates
                        let finalX = (randX - pathCenterX) * scaleFactor;
                        let finalY = (randY - pathCenterY) * scaleFactor;
                        
                        if (selectedGlob.pathIndex === 3) finalX = -finalX;
                        if (selectedGlob.pathIndex === 5) { finalX = -finalX; finalY = -finalY; }

                        d.targetX = finalX;
                        d.targetY = finalY;
                        found = true;
                    }
                    attempts++;
                }
                if(!found) { d.targetX = 0; d.targetY = 0; }
                d.x = 0; d.y = 0; 
            });

            // PHYSICS TWEAKED FOR SPREAD
            simulation.force("x", d3.forceX(d => d.targetX).strength(0.1));
            simulation.force("y", d3.forceY(d => d.targetY).strength(0.1));
            
            // Increased Repulsion to fill gaps
            simulation.force("charge", d3.forceManyBody().strength(-30)); 
            
            // Stronger Collision
            simulation.force("collide", d3.forceCollide().radius(d => priceScale(d.price)).strength(1));
            
            simulation.nodes(activeData).alpha(1).restart();
            simulation.on("tick", () => {
                bubbles.filter(d => d.movement === selectedMovement)
                    .attr("cx", d => d.x)
                    .attr("cy", d => d.y);
                updateLines();
            });
        }

        function resetToPalette() {
            simulation.stop();
            paletteLayer.interrupt(); blobLayer.interrupt(); splatterLayer.interrupt(); bubbles.interrupt();
            dataLayer.selectAll(".connection-line").remove();
            document.getElementById("resetBtn").style.display = "none";
            
            splatterLayer.transition().duration(500).style("opacity", 0).on("end", () => splatterLayer.style("display", "none"));
            bubbles.transition().duration(500).attr("opacity", 0).on("end", function () { d3.select(this).style("display", "none"); });
            
            paletteLayer.transition().duration(800).style("opacity", 1);
            blobLayer.transition().duration(800).style("opacity", 1);
            interactionLayer.style("display", "block"); 
        }

        function handleArtistTrace(event, clickedData) {
            event.stopPropagation(); 
            const artistWorks = data.filter(d => d.artist === clickedData.artist && d.movement === clickedData.movement).sort((a, b) => a.year - b.year);
            bubbles.filter(d => d.style.display !== "none").transition().duration(300).style("opacity", 0.1).style("stroke", "none");
            bubbles.filter(d => d.artist === clickedData.artist && d.movement === clickedData.movement)
                .transition().duration(300).style("opacity", 1).style("stroke", "#333").style("stroke-width", "2px");
            
            dataLayer.selectAll(".connection-line").remove();
            if (artistWorks.length > 1) {
                const p = dataLayer.append("path").datum(artistWorks).attr("class", "connection-line")
                    .attr("d", d3.line().x(d => d.x).y(d => d.y).curve(d3.curveCatmullRom))
                    .style("stroke", "#333").style("stroke-width", 2).style("fill", "none");
                const l = p.node().getTotalLength();
                p.attr("stroke-dasharray", l + " " + l).attr("stroke-dashoffset", l).transition().duration(1500).attr("stroke-dashoffset", 0);
            }
        }

        function updateLines() { 
            const p = dataLayer.select(".connection-line"); 
            if (!p.empty()) { p.attr("d", d3.line().x(d => d.x).y(d => d.y).curve(d3.curveCatmullRom)(p.datum())); }
        }
        
        svg.on("click", function(event) { 
             if (event.target.tagName === "svg" || event.target.classList.contains("splatter-shape")) { 
                bubbles.filter(d => d3.select(event.target).style("display") !== "none")
                    .transition().duration(300).style("opacity", 0.9).style("stroke", "none"); 
                dataLayer.selectAll(".connection-line").remove(); 
            } 
        });

        d3.select("#resetBtn").on("click", resetToPalette);
    });
